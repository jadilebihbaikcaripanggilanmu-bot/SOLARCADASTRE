Setup Supabase DDL 
-- Jalankan di SQL Editor Supabase Anda
-- untuk menyimpan library project:

CREATE TABLE IF NOT EXISTS projects (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  name         TEXT NOT NULL,
  aoi_geojson  JSONB,
  geojson_data JSONB,
  solar_stats  JSONB,
  created_at   TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Aktifkan Row Level Security (RLS)
ALTER TABLE projects ENABLE ROW LEVEL SECURITY;

-- Buat policies akses data user masing-masing
CREATE POLICY "Allow read own projects" ON projects
  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "Allow insert own projects" ON projects
  FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Allow delete own projects" ON projects
  FOR DELETE USING (auth.uid() = user_id);

CREATE POLICY "Allow update own projects" ON projects
  FOR UPDATE USING (auth.uid() = user_id);
